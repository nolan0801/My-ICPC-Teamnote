\begin{itemize}[noitemsep]
    \item $B_n$: Bernoulli numbers
    \item Exponential generating function (EGF): $\frac{x}{e^x-1}=\frac{1}{(e^x-1)/x}=\sum_{n=0}^{\infty}\frac{B_n}{n!}x^n$
    \item General term: $B_n=\sum_{k=0}^{n}\frac{k!(-1)^k}{k+1}S_2(n, k)$
    \item Recurrence relation: $B_0=1;$ $B_n=-\frac{1}{n+1}\sum_{r=0}^{n-1} \binom{n+1}{r}B_r$
    \item Power sum formula: $\sum_{k=1}^{n} k^c=\sum_{k=0}^c \frac{(-1)^k}{c+1}\binom{c+1}{k}B_kn^{c+1-k}$
\end{itemize}
